// Resizable interface

interface Resizable {

void resizeWidth(int width);

void resizeHeight(int height);

}

private int height;

this.height = height;

this.width = width;

}

}


// Method to display the dimensions of the rectangle

System.out.println("Width: " + width + ", Height: " + height);

}

}
